# 命令模式

在讨论命令模式之前，先来实现一个画图(示意)的功能：

```ts
class TV {
  turnOn = () => {
    console.log("电视已打开");
  };

  turnOff = () => {
    console.log("电视已关闭");
  };
}

enum Brightness {
  Low,
  High,
}

class Light {
  brightness: Brightness;

  open = () => {
    if (this.brightness === undefined) {
      this.brightness = Brightness.Low;
      console.log("电灯已打开");
    } else if (this.brightness === Brightness.Low) {
      this.brightness = Brightness.High;
      console.log("电灯已打开");
    } else if (this.brightness === Brightness.High) {
      this.brightness = undefined;
      console.log("电灯已关闭");
    }
  };

  setBrightness = (brightness?: Brightness) => {
    this.brightness = brightness;
  };
}

class RemoteControl {
  private status = "off";
  private undoStack: (string | Brightness)[] = [];
  private redoStack: (string | Brightness)[] = [];
  constructor(private tigger: Light | TV) {}

  onButtonPressed = () => {
    if (this.tigger instanceof Light) {
      this.undoStack.push(this.tigger.brightness);
      this.tigger.open();
    } else if (this.tigger instanceof TV) {
      this.tigger.turnOn();
    }
  };

  undo = () => {
    if (this.tigger instanceof Light) {
      const brightness = this.tigger.brightness;
      if (brightness === Brightness.High) {
        this.tigger.setBrightness(Brightness.Low);
      } else if (brightness === Brightness.Low) {
        this.tigger.setBrightness();
      } else {
      }
    } else if (this.tigger instanceof TV) {
      this.tigger.turnOn();
    }
  };

  redo = () => {
    if (this.status === "off") {
      this.tigger.turnOn();
      this.status = "on";
    } else {
      this.tigger.turnOff();
      this.status = "off";
    }
  };
}

//客户端代码
const testOperateTV = () => {
  const tv = new TV();
  const remote = new RemoteControl(tv);

  remote.onButtonPressed(); // 输出：电视已打开

  // 重做撤销的命令
  remote.redo(); // 输出：电视已关闭
};

//客户端代码
const testOperateLight = () => {
  const light = new Light();
  const turnOnCommand = new TurnOnLightCommand(light);
  const turnOffCommand = new TurnOffLightCommand(light);
  const remote = new RemoteControl();

  // 执行命令
  remote.executeCommand(turnOnCommand); // 输出：电灯已打开
  remote.executeCommand(turnOffCommand); // 输出：电灯已关闭

  // 撤销最后一个命令（关闭电灯）
  remote.undo(); // 输出：电灯已打开

  // 重做撤销的命令
  remote.redo(); // 输出：电灯已关闭
};
```

命令模式

```ts
interface Command {
  execute: () => void;
  undo: () => void;
}

class RemoteControl {
  private undoStack: Command[] = [];
  private redoStack: Command[] = [];

  onButtonPressed = (command: Command) => {
    command.execute();
    this.undoStack.push(command);
    this.redoStack = []; // 清空重做栈，因为新命令改变了状态
  };

  undo = () => {
    if (this.undoStack.length > 0) {
      const command = this.undoStack.pop();
      command.undo();
      this.redoStack.push(command);
    }
  };

  redo = () => {
    if (this.redoStack.length > 0) {
      const command = this.redoStack.pop();
      command.execute();
      this.undoStack.push(command);
    }
  };
}

class TV {
  turnOn = () => {
    console.log("电视已打开");
  };

  turnOff = () => {
    console.log("电视已关闭");
  };
}

class Light {
  turnOn = () => {
    console.log("电灯已打开");
  };

  turnOff = () => {
    console.log("电灯已关闭");
  };

class TurnOnTVCommand implements Command {
  constructor(private tv: TV) {}

  execute = () => {
    this.tv.turnOn();
  };

  undo = () => {
    this.tv.turnOff();
  };
}

class TurnOffTVCommand implements Command {
  constructor(private tv: TV) {}

  execute = () => {
    this.tv.turnOff();
  };

  undo = () => {
    this.tv.turnOn();
  };
}
```
