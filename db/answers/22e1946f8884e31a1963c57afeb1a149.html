// 参考自 segmentfault 的@大笑平
const currying = (fn) =>
  (judge = (...args) =>
    args.length >= fn.length
      ? fn(...args)
      : (...arg) => judge(...args, ...arg));

// Test
const fn = currying(function(a, b, c) {
  console.log([a, b, c]);
});

function add(x, y, z) {
  return x + y + z;
}
