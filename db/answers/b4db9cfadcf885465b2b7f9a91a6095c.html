  const concatLiContent = (ul) => {
    const lis = ul.children;
    let str='';
    for(let i = 0; i < lis.length; i++) {
      const li = lis[i];
      str+=li.innerText;
    }
    return str;
  }
  或
  const concatLiContent = (ul) => {
    const lis = ul.children;
    const realArr = [].slice.call(lis);
    let str='';
    realArr.forEach((li) => {
      str+=li.innerText;
    })
    return str;
  }
  const ul = document.getElementById('ul');
  concatLiContent(ul);
